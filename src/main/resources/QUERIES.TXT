список карт клиента

SELECT LOGIN, CARD.NUMBER FROM CLIENT, CARD, ACCOUNT, CLIENTS_AND_ACCOUNTS, ACCOUNTS_AND_CARDS
WHERE CLIENT.ID = CLIENTS_AND_ACCOUNTS.CLIENT_ID AND
      ACCOUNT.ID = CLIENTS_AND_ACCOUNTS.ACCOUNT_ID AND
      CARD.ID = ACCOUNTS_AND_CARDS.CARD_ID AND
      ACCOUNT.ID = ACCOUNTS_AND_CARDS.ACCOUNT_ID AND
      CLIENT.LOGIN = 'RITA';
